SELECT ROW_NUMBER() OVER(ORDER BY NOTICE_NO DESC) AS NUM, NOTICE_TBL.* FROM NOTICE_TBL;

문법이 낯설다.

NUM을 기준으로 WHERE 조건절로 바로 사용할 수 없고 
인라인뷰를 써야한다.

------------------------------
SELECT *
FROM(
SELECT ROW_NUMBER() OVER(ORDER BY NOTICE_NO DESC) AS NUM, NOTICE_TBL.* FROM NOTICE_TBL)
WHERE NUM BETWEEN 6 AND 10;
